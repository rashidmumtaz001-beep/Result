// Audio synthesizer cue engine
function playToneSound(freq, type = 'sine', duration = 0.1) {
    try {
        const synth = new Tone.Synth({
            oscillator: { type: type },
            envelope: { attack: 0.01, decay: 0.1, sustain: 0.2, release: 0.1 }
        }).toDestination();
        synth.triggerAttackRelease(freq, duration);
    } catch(e) {
        try {
            const ctx = new (window.AudioContext || window.webkitAudioContext)();
            const osc = ctx.createOscillator();
            const gain = ctx.createGain();
            osc.connect(gain);
            gain.connect(ctx.destination);
            osc.type = type;
            osc.frequency.setValueAtTime(freq, ctx.currentTime);
            gain.gain.setValueAtTime(0.08, ctx.currentTime);
            osc.start();
            osc.stop(ctx.currentTime + duration);
        } catch (err) {}
    }
}


// Toast notifications
function showToast(title, message, type = 'success') {
    const container = document.getElementById('toastContainer');
    if (!container) return;

    let bgClass = "bg-white dark:bg-slate-950 border-emerald-500/50";
    let textClass = "text-emerald-500";
    let iconClass = "fa-circle-check";
    
    if (type === 'error') {
        bgClass = "bg-white dark:bg-slate-950 border-rose-500/50";
        textClass = "text-rose-500";
        iconClass = "fa-circle-xmark";
    } else if (type === 'info') {
        bgClass = "bg-white dark:bg-slate-950 border-brand-500/50";
        textClass = "text-brand-500";
        iconClass = "fa-circle-info";
    }

    const toast = document.createElement('div');
    toast.className = `flex items-center gap-3 p-4 rounded-xl border-l-4 ${bgClass} shadow-xl animate-fadeIn transition-all duration-300`;
    toast.innerHTML = `
        <div class="text-lg ${textClass} shrink-0">
            <i class="fa-solid ${iconClass}"></i>
        </div>
        <div>
            <h4 class="font-bold text-xs text-slate-800 dark:text-slate-100">${title}</h4>
            <p class="text-[10px] text-slate-400 mt-0.5">${message}</p>
        </div>
    `;
    
    container.appendChild(toast);
    playToneSound(type === 'error' ? 220 : (type === 'info' ? 523 : 880), 'sine', 0.12);

    setTimeout(() => {
        toast.classList.add('opacity-0');
        setTimeout(() => toast.remove(), 300);
    }, 4000);
}

// Switch Tabs Panel Routing
function switchTab(tabName) {
    document.querySelectorAll('.tab-content').forEach(element => {
        element.classList.add('hidden');
    });
    document.querySelectorAll('.tab-link').forEach(link => {
        link.classList.remove('bg-brand-700', 'text-white', 'shadow-md');
        link.classList.add('text-brand-100', 'hover:bg-brand-700/40', 'hover:text-white');
    });

    const targetSection = document.getElementById('tab-' + tabName);
    if (targetSection) targetSection.classList.remove('hidden');

    const activeLink = document.querySelector(`.tab-link[data-tab="${tabName}"]`);
    if (activeLink) {
        activeLink.classList.remove('text-brand-100', 'hover:bg-brand-700/40', 'hover:text-white');
        activeLink.classList.add('bg-brand-700', 'text-white', 'shadow-md');
    }

    const titleElement = document.getElementById('currentTabTitle');
    if (titleElement) {
        let text = "Dashboard Hub";
        let icon = "fa-chart-line";
        
        switch (tabName) {
            case 'students': text = "Student Base Directory"; icon = "fa-user-graduate"; break;
            case 'staff': text = "Faculty & Staff Directory"; icon = "fa-chalkboard-user"; break;
            case 'fees': text = "Fee Ledger Registry"; icon = "fa-file-invoice-dollar"; break;
            case 'classes': text = "Class Timetables Schedule"; icon = "fa-school"; break;
            case 'attendance': text = "Daily Attendance Log"; icon = "fa-calendar-check"; break;
            case 'exams': text = "Examination Performance Ledger"; icon = "fa-square-poll-vertical"; break;
            case 'reports': text = "Analytical Intelligence Reports"; icon = "fa-chart-pie"; break;
            case 'whatsapp': text = "WhatsApp Notification Engine"; icon = "fa-whatsapp text-emerald-500"; break;
            case 'settings': text = "System Parameters Configuration"; icon = "fa-sliders"; break;
        }
        titleElement.innerHTML = `<i class="fa-solid ${icon}"></i> ${text}`;
    }

    const sidebarMenu = document.getElementById('sidebarMenu');
    if (window.innerWidth < 768 && sidebarMenu) {
        sidebarMenu.classList.add('hidden');
    }

    // Dynamic Tab Load triggers
    if (tabName === 'dashboard') {
        renderDashboardSummary();
    } else if (tabName === 'students') {
        renderStudentsTable();
    } else if (tabName === 'staff') {
        renderStaffRoster();
    } else if (tabName === 'fees') {
        renderFeeLedger();
    } else if (tabName === 'classes') {
        renderClassesAndTimetable();
    } else if (tabName === 'attendance') {
        loadAttendanceSheet();
    } else if (tabName === 'exams') {
        renderExamsLedger();
    } else if (tabName === 'reports') {
        renderReportsCharts();
    } else if (tabName === 'whatsapp') {
        selectWhatsAppTemplate('fee');
    }
}

function renderDashboardSummary() {

function renderClassAdmissionsDistribution() {
    const parent = document.getElementById('classProgressBars');
    if (!parent) return;
    
    parent.innerHTML = "";
    const counts = {};
    appState.students.forEach(s => {
        counts[s.className] = (counts[s.className] || 0) + 1;
    });

    const coreClasses = ['ICS', 'FSc', '10th', '1st'];
    coreClasses.forEach(grp => {
        const cnt = counts[grp] || 0;
        const pct = appState.students.length > 0 ? Math.round((cnt / appState.students.length) * 100) : 0;
        
        parent.innerHTML += `
            <div class="p-4 bg-slate-50 dark:bg-slate-900 border border-slate-150 dark:border-slate-800 rounded-xl">
                <div class="flex justify-between items-center text-xs font-semibold mb-1">
                    <span class="text-slate-700 dark:text-slate-300">Grade Class ${grp}</span>
                    <span class="text-brand-600 dark:text-brand-400 font-bold">${cnt} Students</span>
                </div>
                <div class="w-full bg-slate-200 dark:bg-slate-850 h-2 rounded-full overflow-hidden">
                    <div class="bg-gradient-to-r from-brand-500 to-indigo-600 h-2 rounded-full" style="width: ${pct}%"></div>
                </div>
            </div>
        `;
    });
}

// Fill student dropdown selector options across the suite
function populateStudentSelectors() {
    const selectors = ['feeStudentSelector', 'gradeStudentSelector', 'quickTestStudent'];
    
    selectors.forEach(id => {
        const dropdown = document.getElementById(id);
        if (!dropdown) return;
        
        dropdown.innerHTML = "";
        if (appState.students.length === 0) {
            dropdown.innerHTML = `<option value="">No enrolled students</option>`;
            return;
        }

        appState.students.forEach(stud => {
            const option = document.createElement('option');
            if (id === 'gradeStudentSelector') {
                option.value = `${stud.name}:${stud.className}`;
            } else {
                option.value = stud.id;
            }
            option.textContent = `${stud.name} (${stud.id} - ${stud.className})`;
            dropdown.appendChild(option);
        });
    });
}

// Live grade calculation preview
function updateLiveGradePreview() {
    const obtainedInput = parseFloat(document.getElementById('quickTestObtained').value) || 0;
    const totalInput = parseFloat(document.getElementById('quickTestTotal').value) || 100;
    
    let percentage = 0;
    if (totalInput > 0) {
        percentage = Math.round((obtainedInput / totalInput) * 100);
    }
    
    // Keep range boundaries
    percentage = Math.max(0, Math.min(percentage, 100));
    
    // Determine rating & pass / fail
    let grade = 'F';
    let status = 'FAIL';
    let ratingColorClass = 'text-rose-600 dark:text-rose-400';
    let bgStatusClass = 'bg-rose-100 dark:bg-rose-950/30 text-rose-600 dark:text-rose-400';
    let circleColor = '#f43f5e'; // rose

    if (percentage >= 85) {
        grade = 'A+';
        circleColor = '#10b981'; // emerald
        ratingColorClass = 'text-emerald-600 dark:text-emerald-400';
        bgStatusClass = 'bg-emerald-100 dark:bg-emerald-950/30 text-emerald-600 dark:text-emerald-400';
        status = 'PASS';
    } else if (percentage >= 75) {
        grade = 'A';
        circleColor = '#34d399'; // light emerald
        ratingColorClass = 'text-emerald-500 dark:text-emerald-400';
        bgStatusClass = 'bg-emerald-100 dark:bg-emerald-950/30 text-emerald-500 dark:text-emerald-400';
        status = 'PASS';
    } else if (percentage >= 60) {
        grade = 'B';
        circleColor = '#3b82f6'; // blue
        ratingColorClass = 'text-brand-500 dark:text-brand-400';
        bgStatusClass = 'bg-brand-100 dark:bg-brand-950/30 text-brand-500 dark:text-brand-400';
        status = 'PASS';
    } else if (percentage >= 45) {
        grade = 'C';
        circleColor = '#f59e0b'; // amber
        ratingColorClass = 'text-amber-500 dark:text-amber-400';
        bgStatusClass = 'bg-amber-100 dark:bg-amber-950/30 text-amber-500 dark:text-amber-400';
        status = 'PASS';
    } else if (percentage >= 33) {
        grade = 'D';
        circleColor = '#f97316'; // orange
        ratingColorClass = 'text-orange-500';
        bgStatusClass = 'bg-orange-100 dark:bg-orange-950/30 text-orange-500';
        status = 'PASS';
    }

    // Update gauge visual items
    document.getElementById('liveGradePercent').innerText = `${percentage}%`;
    document.getElementById('liveGradeRating').innerText = `GRADE ${grade}`;
    document.getElementById('liveGradeRating').className = `text-[10px] font-bold tracking-wider ${ratingColorClass}`;
    
    const statusBadge = document.getElementById('liveGradeStatus');
    statusBadge.innerText = status;
    statusBadge.className = `px-2.5 py-0.5 rounded-full text-[9px] font-extrabold uppercase ${bgStatusClass}`;
    
    // Modify radial progress circle dashoffset
    // Perimeter = 2 * PI * r (r=40 -> ~251.2)
    const perimeter = 251.2;
    const offset = perimeter - (percentage / 100) * perimeter;
    
    const ring = document.getElementById('liveGradeRing');
    ring.setAttribute('stroke-dashoffset', offset);
    ring.style.color = circleColor;
}

// Quick test record submission on dashboard
function handleQuickTestSubmit(e) {
    e.preventDefault();
    const studId = document.getElementById('quickTestStudent').value;
    const studObj = appState.students.find(s => s.id === studId);
    if (!studObj) {
        showToast("Invalid Student", "Select or register a student first.", "error");
        return;
    }

    const obtained = parseFloat(document.getElementById('quickTestObtained').value);
    const total = parseFloat(document.getElementById('quickTestTotal').value);

    if (obtained > total) {
        showToast("Marks Error", "Obtained marks cannot exceed total marks limit.", "error");
        return;
    }

    const newExam = {
        studentName: studObj.name,
        className: studObj.className,
        subject: document.getElementById('quickTestSubject').value.trim(),
        obtained: obtained,
        total: total
    };

    appState.exams.push(newExam);
    saveLocalCache();
    showToast("Result Registered!", `Test score saved for ${studObj.name}`, "success");
    logActivity(`Posted exam result (${obtained}/${total}) for student ${studObj.name} via dashboard.`);
    
    document.getElementById('quickTestRecordForm').reset();
    
    // Refresh gauges and states
    updateLiveGradePreview();
    renderDashboardSummary();
}

function renderExamsLedger() {
    const tableBody = document.getElementById('gradeTableBody');
    if (!tableBody) return;

    tableBody.innerHTML = "";
    if (appState.exams.length === 0) {
        tableBody.innerHTML = `<tr><td colspan="6" class="text-center py-8 text-slate-400">No student exam marks recorded in performance ledger.</td></tr>`;
    } else {
        appState.exams.forEach((exam, index) => {
            const pct = Math.round((parseFloat(exam.obtained) / parseFloat(exam.total)) * 100);
            
            let grade = 'F';
            let color = 'text-rose-500';
            if (pct >= 85) { grade = 'A+'; color = 'text-emerald-500'; }
            else if (pct >= 75) { grade = 'A'; color = 'text-emerald-400'; }
            else if (pct >= 60) { grade = 'B'; color = 'text-brand-500'; }
            else if (pct >= 45) { grade = 'C'; color = 'text-amber-500'; }
            else if (pct >= 33) { grade = 'D'; color = 'text-orange-500'; }

            tableBody.innerHTML += `
                <tr class="hover:bg-slate-50/60 dark:hover:bg-slate-900/40 transition-all">
                    <td class="p-3 font-semibold text-slate-800 dark:text-slate-100">${exam.studentName}</td>
                    <td class="p-3"><span class="px-2 py-0.5 rounded bg-slate-100 dark:bg-slate-800 font-bold text-[10px] text-slate-600 dark:text-slate-300">Class ${exam.className}</span></td>
                    <td class="p-3 text-slate-500">${exam.subject}</td>
                    <td class="p-3 text-center font-bold font-mono text-slate-700 dark:text-slate-200">${exam.obtained} / ${exam.total} <span class="text-slate-400 font-normal text-[10px]">(${pct}%)</span></td>
                    <td class="p-3 text-center font-bold text-sm ${color}">${grade}</td>
                    <td class="p-3 text-right">
                        <button onclick="deleteExamRecord(${index})" class="text-rose-500 hover:text-rose-600 transition-all font-semibold" title="Discard result record">
                            <i class="fa-solid fa-square-minus"></i>
                        </button>
                    </td>
                </tr>
            `;
        });
    }
}

function saveAttendanceRoster() {
    const grp = document.getElementById('attendanceClassFilter').value;
    const dateVal = document.getElementById('attendanceDatePicker').value;
    
    if (!appState.attendance[dateVal]) {
        appState.attendance[dateVal] = {};
    }

    const matchingStudents = appState.students.filter(s => s.className === grp);
    matchingStudents.forEach(student => {
        const checkedRadio = document.querySelector(`input[name="att-${student.id}"]:checked`);
        if (checkedRadio) {
            appState.attendance[dateVal][student.id] = checkedRadio.value;
        }
    });

    saveLocalCache();
    showToast("Attendance Committed!", `Roll call registered for Class ${grp} date ${dateVal}`, "success");
    logActivity("Committed roll call log for Class Group " + grp);
}

function renderReportsCharts() {
    const gradeCanvas = document.getElementById('gradeBreakdownChart');
    const billingCanvas = document.getElementById('collectionsPieChart');
    if (!gradeCanvas || !billingCanvas) return;

    if (gradeBreakdownChartInstance) gradeBreakdownChartInstance.destroy();
    if (collectionsPieChartInstance) collectionsPieChartInstance.destroy();

    let countA = 0;
    let countB = 0;
    let countPassing = 0;
    let countFailing = 0;

    appState.exams.forEach(ex => {
        const pct = (parseFloat(ex.obtained) / parseFloat(ex.total)) * 100;
        if (pct >= 75) countA++;
        else if (pct >= 60) countB++;
        else if (pct >= 33) countPassing++;
        else countFailing++;
    });

    gradeBreakdownChartInstance = new Chart(gradeCanvas, {
        type: 'doughnut',
        data: {
            labels: ['A/A+ Grade', 'B Grade', 'Passing Range', 'Failing Range'],
            datasets: [{
                data: [countA || 2, countB || 1, countPassing || 1, countFailing || 0],
                backgroundColor: ['#10b981', '#3b5ee7', '#f59e0b', '#ef4444'],
                borderWidth: 0
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            plugins: {
                legend: { position: 'bottom', labels: { boxWidth: 12, font: { size: 10 } } }
            }
        }
    });

    let totalCashReceived = 0;
    appState.feeLedger.forEach(txn => {
        totalCashReceived += parseFloat(txn.amount);
    });
    
    const totalUnpaidStudents = appState.students.filter(s => s.feeStatus === 'Unpaid').length;
    let totalOutstandingAmount = totalUnpaidStudents * 5000;

    collectionsPieChartInstance = new Chart(billingCanvas, {
        type: 'pie',
        data: {
            labels: ['Collected Inflow (PKR)', 'Outstanding Dues (PKR)'],
            datasets: [{
                data: [totalCashReceived || 12000, totalOutstandingAmount || 25000],
                backgroundColor: ['#10b981', '#f43f5e'],
                borderWidth: 0
            }]
        },
        options: {
            responsive: true,
            maintainAspectRatio: false,
            plugins: {
                legend: { position: 'bottom', labels: { boxWidth: 12, font: { size: 10 } } }
            }
        }
    });
}


function selectWhatsAppTemplate(type) {
    document.getElementById('template-btn-fee').className = "p-3 rounded-xl border border-slate-200 dark:border-slate-800 text-left transition-all";
    document.getElementById('template-btn-absent').className = "p-3 rounded-xl border border-slate-200 dark:border-slate-800 text-left transition-all";
    document.getElementById('template-btn-custom').className = "p-3 rounded-xl border border-slate-200 dark:border-slate-800 text-left transition-all";

    const targetBtn = document.getElementById('template-btn-' + type);
    if (targetBtn) {
        targetBtn.className = "p-3 rounded-xl border-2 border-brand-500 bg-brand-50/50 dark:bg-brand-950/20 text-left transition-all relative";
    }

    const preview = document.getElementById('waMessageText');
    if (!preview) return;

    if (type === 'fee') {
        preview.value = `Dear Parent, school tuition fee remains outstanding. Kindly clear dues as soon as possible. - Registrar Office, ${appState.academyName}`;

function launchRealWhatsAppWeb() {
    const num = document.getElementById('waRecipient').value.trim();
    const text = encodeURIComponent(document.getElementById('waMessageText').value.trim());

    if (!num) {
        showToast("Error", "Kindly insert parent guardian WhatsApp contact phone number.", "error");
        return;
    }

    const cleanedNum = num.replace(/\D/g, ''); 
    const url = `https://api.whatsapp.com/send?phone=${cleanedNum}&text=${text}`;
    
    showToast("Launching Link...", "Opening API transmission browser link.", "info");
    logActivity(`Launched WhatsApp Redirect Web Hook for number: ${cleanedNum}`);
    window.open(url, '_blank');
}


function saveAcademyConfiguration() {
    const name = document.getElementById('cfgAcademyName').value.trim();
    const email = document.getElementById('cfgAcademyEmail').value.trim();
    const curr = document.getElementById('cfgAcademyCurrency').value.trim();

    if (!name || !email) {
        showToast("Form Validation Error", "Academy Name and administrative email cannot be blank.", "error");
        return;
    }

    appState.academyName = name;
    appState.academyEmail = email;
    appState.currency = curr;

    saveLocalCache();
    showToast("System Config Saved", "Administrative parameters written to cache memory.", "success");
    logActivity(`Changed global academy configuration profile to ${name}`);
}
